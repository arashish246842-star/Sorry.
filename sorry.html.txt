  <script>
    // slideshow
    const fileInput = document.getElementById('fileInput');
    const mainImage = document.getElementById('mainImage');
    const thumbs = document.getElementById('thumbs');
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');

    let images = [
      'https://via.placeholder.com/900x520.png?text=Photo+1',
      'https://via.placeholder.com/900x520.png?text=Photo+2',
      'https://via.placeholder.com/900x520.png?text=Photo+3'
    ];
    let current = 0;
    let slideInterval = null;

    function refreshThumbnails(){
      thumbs.innerHTML = '';
      images.forEach((src,i)=>{
        const d = document.createElement('div');
        d.className = 'thumb' + (i===current ? ' active' : '');
        const img = document.createElement('img');
        img.src = src;
        d.appendChild(img);
        d.onclick = ()=> { showIndex(i); stopAuto(); }
        thumbs.appendChild(d);
      });
    }

    function showIndex(i){
      current = (i + images.length) % images.length;
      mainImage.src = images[current];
      refreshThumbnails();
    }

    function next(){
      showIndex(current+1);
    }
    function prev(){
      showIndex(current-1);
    }

    prevBtn.onclick = ()=> { stopAuto(); prev(); }
    nextBtn.onclick = ()=> { stopAuto(); next(); }

    function startAuto(){
      stopAuto();
      slideInterval = setInterval(next, 3500);
    }
    function stopAuto(){ if(slideInterval) { clearInterval(slideInterval); slideInterval=null } }

    fileInput.addEventListener('change', (e)=>{
      const files = Array.from(e.target.files).filter(f => f.type.startsWith('image/')).slice(0,8);
      if(files.length === 0) return;
      images = [];
      const promises = files.map(f => new Promise((res,rej)=>{
        const r = new FileReader();
        r.onload = ()=> res(r.result);
        r.onerror = rej;
        r.readAsDataURL(f);
      }));
      Promise.all(promises).then(dataUrls=>{
        images = dataUrls;
        current = 0;
        showIndex(0);
        startAuto();
      });
    });

    refreshThumbnails();
    startAuto();

    // popup
    const popup = document.getElementById('popup');
    const forgiveBtn = document.getElementById('forgiveBtn') || document.getElementById('forgiveBtn');
    const closePopup = document.getElementById('closePopup');

    document.getElementById('forgiveBtn').addEventListener('click', ()=>{
      popup.style.transform = 'translate(-50%,-50%) scale(1)';
      popup.setAttribute('aria-hidden','false');
    });
    closePopup.addEventListener('click', ()=>{
      popup.style.transform = 'translate(-50%,-50%) scale(0)';
      popup.setAttribute('aria-hidden','true');
    });

    // petals
    function generatePetals(n=18){
      for(let i=0;i<n;i++){
        const p = document.createElement('div');
        p.className='petal';
        p.style.left = Math.random()*100 + 'vw';
        p.style.top = (Math.random()*-20) + 'vh';
        p.style.animationDuration = (6 + Math.random()*6) + 's';
        p.style.transform = `rotate(${Math.random()*360}deg)`;
        p.style.opacity = 0.8 + Math.random()*0.2;
        document.body.appendChild(p);
        setTimeout(()=> p.remove(), 16000 + Math.random()*8000);
      }
      setTimeout(()=> generatePetals(8), 9000 + Math.random()*5000);
    }
    generatePetals(20);

    // YouTube controls (reload iframe to request autoplay when user taps)
    const playSongBtn = document.getElementById('playSong');
    const pauseSongBtn = document.getElementById('pauseSong');
    const yt = document.getElementById('ytplayer');

    playSongBtn.addEventListener('click', ()=>{
      yt.src = "https://www.youtube.com/embed/kyjg5kX4pT0?si=gM4fP1lLm9sd4CPB&autoplay=1&loop=1&playlist=kyjg5kX4pT0&rel=0";
    });
    pauseSongBtn.addEventListener('click', ()=>{
      yt.src = '';
    });

    // keyboard nav
    document.addEventListener('keydown', (e)=>{
      if(e.key === 'ArrowRight') { next(); stopAuto() }
      if(e.key === 'ArrowLeft') { prev(); stopAuto() }
      if(e.key === ' ') { if(slideInterval) stopAuto(); else startAuto(); }
    });
  </script>
